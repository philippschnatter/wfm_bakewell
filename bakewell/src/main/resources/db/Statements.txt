CREATE TABLE IF NOT EXISTS CUSTOMER(
ID INT PRIMARY KEY,
FIRSTNAME VARCHAR(255),
LASTNAME VARCHAR(255),
ADDRESS VARCHAR(255),
TELNO VARCHAR(255),
MAILADDRESS VARCHAR(255),
COMPANY VARCHAR(255),
PASSWORD VARCHAR(255)
);

CREATE TABLE IF NOT EXISTS INGREDIENT(
ID INT PRIMARY KEY,
NAME VARCHAR(255),
AMOUNT DOUBLE,
FAT DOUBLE,
SUGAR DOUBLE,
CALORIES DOUBLE,
PROTEIN VARCHAR(255)
);

CREATE TABLE IF NOT EXISTS RECIPE(
ID INT PRIMARY KEY,
NAME VARCHAR(255)
);

CREATE TABLE IF NOT EXISTS PRODUCT(
ID INT PRIMARY KEY,
RECIPE_ID INT,
CUSTOMER_ID INT,
PRODUCTNAME VARCHAR(255),
DELIVERYDATE DATE,
PRODUCTION_START DATE,
PRODUCTION_END DATE,
PRODUCTION_FACILITY VARCHAR(255),
PACKAGING_CONTRACTOR VARCHAR(255),
TRANSPORT_COST DOUBLE,
FOREIGN KEY (RECIPE_ID) REFERENCES RECIPE(ID) ON UPDATE CASCADE ON DELETE CASCADE,
FOREIGN KEY (CUSTOMER_ID) REFERENCES CUSTOMER(ID) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS  INGREDIENT2RECIPE(
ID INT PRIMARY KEY,
INGREDIENT_ID INT,
RECIPE_ID INT,
AMOUNT DOUBLE,
GDA_CALORIES DOUBLE,
GDA_FAT DOUBLE,
GDA_SUGAR DOUBLE,
GDA_PROTEIN DOUBLE,
FOREIGN KEY (INGREDIENT_ID) REFERENCES INGREDIENT(ID) ON UPDATE CASCADE ON DELETE CASCADE,
FOREIGN KEY (RECIPE_ID) REFERENCES RECIPE(ID) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE SEQUENCE IF NOT EXISTS ID_CUSTOMER_SEQ START WITH 1; 
CREATE SEQUENCE IF NOT EXISTS ID_INGREDIENT_SEQ START WITH 1;
CREATE SEQUENCE IF NOT EXISTS ID_PRODUCT_SEQ START WITH 1;
CREATE SEQUENCE IF NOT EXISTS ID_RECIPE_SEQ START WITH 1;
CREATE SEQUENCE IF NOT EXISTS ID_INGREDIENT2RECIPE_SEQ START WITH 1;

DROP TABLE CUSTOMER;
DROP TABLE INGREDIENT;
DROP TABLE PRODUCT;
DROP TABLE RECIPE;
DROP TABLE INGREDIENT2RECIPE;
DROP SEQUENCE ID_CUSTOMER_SEQ;
DROP SEQUENCE ID_INGREDIENT_SEQ;
DROP SEQUENCE ID_PRODUCT_SEQ;
