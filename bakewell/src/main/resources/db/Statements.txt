CREATE TABLE IF NOT EXISTS CUSTOMER(
ID INT PRIMARY KEY,
FIRSTNAME VARCHAR(255),
LASTNAME VARCHAR(255),
ADDRESS VARCHAR(255),
TELNO VARCHAR(255),
MAILADDRESS VARCHAR(255),
COMPANY VARCHAR(255),
PASSWORD VARCHAR(255)
);

CREATE TABLE IF NOT EXISTS INGREDIENT(
ID INT PRIMARY KEY,
NAME VARCHAR(255),
GDA_ENERGY DOUBLE,
GDA_PROTEIN DOUBLE,
GDA_CARBO DOUBLE,
GDA_FAT DOUBLE,
GDA_FIBER DOUBLE,
GDA_SODIUM DOUBLE,
PRICE DOUBLE
);

CREATE TABLE IF NOT EXISTS RECIPE(
ID INT PRIMARY KEY,
NAME VARCHAR(255),
DESCRIPTION VARCHAR(MAX),
ALLGDA_ENERGY DOUBLE,
ALLGDA_PROTEIN DOUBLE,
ALLGDA_CARBO DOUBLE,
ALLGDA_FAT DOUBLE,
ALLGDA_FIBER DOUBLE,
ALLGDA_SODIUM DOUBLE,
TOTALPRICE DOUBLE
);

CREATE TABLE IF NOT EXISTS PRODUCT(
ID INT PRIMARY KEY,
RECIPE_ID INT,
CUSTOMER_ID INT,
PRODUCTNAME VARCHAR(255),
DELIVERYDATE DATE,
PRODUCTION_START DATE,
PRODUCTION_END DATE,
PRODUCTION_FACILITY VARCHAR(255),
PRODUCTION_CONTRACTOR VARCHAR(255),
TRANSPORT_CONTRACTOR VARCHAR(255),
TRANSPORT_COST DOUBLE,
FOREIGN KEY (RECIPE_ID) REFERENCES RECIPE(ID) ON UPDATE CASCADE ON DELETE CASCADE,
FOREIGN KEY (CUSTOMER_ID) REFERENCES CUSTOMER(ID) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS INGREDIENT2RECIPE(
ID INT PRIMARY KEY,
RECIPE_ID INT,
INGREDIENT_ID INT,
AMOUNT DOUBLE,
FOREIGN KEY (RECIPE_ID) REFERENCES RECIPE(ID) ON UPDATE CASCADE ON DELETE CASCADE,
FOREIGN KEY (INGREDIENT_ID) REFERENCES INGREDIENT(ID) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE GDAREFERENCE(
ID INT PRIMARY KEY,
NAME VARCHAR(255),
AMOUNT DOUBLE
);
